def read_csv_first_n(csv_path, n=10):
    """Read first n rows of a CSV file."""
    rows = []
    with open(csv_path, 'r') as f:
        reader = csv.DictReader(f)
        for i, row in enumerate(reader):
            if i >= n:
                break
            rows.append(row)
    return rows

def main():
