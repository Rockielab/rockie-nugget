Prepare evaluation JSONL files from the CSV dataset.
For the first 10 rows, match to reference by submission_id_v0,
then create two JSONL files (for v0 and v1 eval).
"""
import pandas as pd
import json

# Read first 10 rows of CSV
df = pd.read_csv("data/sample/generations.csv", nrows=10)
print(f"Loaded {len(df)} rows")
print(f"Columns: {list(df.columns)}")

# Read reference file
refs = []
with open("data/sample/py_reference.jsonl") as f:
    for line in f:
        refs.append(json.loads(line))

# Build lookup by submission_id_v0
ref_by_sid = {}
for r in refs:
    ref_by_sid[r["submission_id_v0"]] = r

# Prepare v0 and v1 entries
v0_entries = []
v1_entries = []

for _, row in df.iterrows():
    sid = row["submission_id_v0"]
    pid = row["problem_id"]
    lang = row["language"]
    
    # Find reference
    ref = ref_by_sid.get(sid)
    if ref is None:
        print(f"WARNING: no reference for {sid}, problem {pid}")
        continue
    
    # Build entry for v0
    v0_entry = {
        "problem_id": pid,
        "submission_id_v0": sid,
        "language": lang,
        "input": ref["input"],
        "generated_answer": row["code_v0"],
        "target": ref["target"]
    }
    v0_entries.append(v0_entry)
    
    # Build entry for v1
    v1_entry = {
        "problem_id": pid,
        "submission_id_v0": sid,
        "language": lang,
        "input": ref["input"],
        "generated_answer": row["code_v1"],
        "target": ref["target"]
    }
    v1_entries.append(v1_entry)

print(f"Prepared {len(v0_entries)} v0 entries, {len(v1_entries)} v1 entries")

# Write JSONL
with open("data/sample/v0_gen.jsonl", "w") as f:
    for entry in v0_entries:
        f.write(json.dumps(entry) + "\n")

with open("data/sample/v1_gen.jsonl", "w") as f:
    for entry in v1_entries:
        f.write(json.dumps(entry) + "\n")

print("Wrote v0_gen.jsonl and v1_gen.jsonl")

# Also report the problem_ids
print("\nProblem IDs in eval set:")
for e in v0_entries:
    print(f"  {e['problem_id']} (sid={e['submission_id_v0']}, lang={e['language']})")

# Report unique problem_ids
unique_pids = set(e["problem_id"] for e in v0_entries)
unique_langs = set(e["language"] for e in v0_entries)
print(f"\nUnique problem_ids: {unique_pids}")
print(f"Unique languages: {unique_langs}")
