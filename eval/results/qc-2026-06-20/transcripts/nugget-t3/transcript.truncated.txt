"""Simple evaluation script for pie-perf benchmark.
Reads generations.csv, takes first 10 rows, runs code_v0 and code_v1 against public test cases,
and reports input_acc for each row."""

import csv
import os
import subprocess
import tempfile
import shutil
import json
import glob
import time
import sys

def read_first_10_rows(csv_path):
    rows = []
    with open(csv_path, 'r') as f:
        reader = csv.DictReader(f)
        for i, row in enumerate(reader):
            if i >= 10:
                break
            rows.append(row)
    return rows

def compile_cpp(code, exe_path):
    """Compile C++ code, returns True on success."""
    src_path = exe_path + ".cpp"
    with open(src_path, 'w') as f:
        f.write(code)
    try:
        result = subprocess.run(
            ['g++', '--std=c++17', '-O1', src_path, '-o', exe_path],
            capture_output=True, text=True, timeout=30
        )
        if result.returncode != 0:
            print(f"  Compile error: {result.stderr[:200]}")
            return False
        return True
    except Exception as e:
        print(f"  Compile exception: {e}")
        return False

def run_program(exe_path, input_path, timeout=10):
    """Run compiled program with input from file, return (stdout, stderr, exit_code, timed_out)."""
    try:
        with open(input_path, 'r') as f:
            result = subprocess.run(
                [exe_path],
                stdin=f,
                capture_output=True, text=True,
                timeout=timeout
            )
        return result.stdout.strip(), result.stderr.strip(), result.returncode, False
    except subprocess.TimeoutExpired:
        return "", "TIMEOUT", -1, True

def evaluate_code(code, language, problem_id, test_cases_base, num_trials=2):
    """Evaluate code against test cases, return accuracy."""
    if language.lower() == 'c++' or language.lower() == 'cpp':
        exe_path = os.path.join(tempfile.gettempdir(), f"eval_{problem_id}_{os.getpid()}_{id(code)}")
        if not compile_cpp(code, exe_path):
            return 0.0
        
        test_dir = os.path.join(test_cases_base, problem_id)
        # find all input files
        input_files = sorted(glob.glob(os.path.join(test_dir, "input.*.txt")))
        output_files = sorted(glob.glob(os.path.join(test_dir, "output.*.txt")))
        
        if not input_files:
            print(f"  No test cases found for {problem_id}")
            # Cleanup
            for f in [exe_path, exe_path + ".cpp"]:
                if os.path.exists(f):
                    os.remove(f)
            return 0.0
